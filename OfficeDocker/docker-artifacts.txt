# AxQxOS Kernel Runner — Agent Q Digital Twin
# Base: Python 3.12 slim + Node 20 for MCP servers
FROM python:3.12-slim AS base

LABEL org.axqxos.agent="AgentQ"
LABEL org.axqxos.epoch="Q5"
LABEL org.axqxos.sovereignty="RFC8785"

# System deps
RUN apt-get update && apt-get install -y --no-install-recommends \
    git curl ca-certificates build-essential nodejs npm \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /axqxos

# Python deps
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Node deps (MCP servers)
COPY package.json package-lock.json ./
RUN npm ci --production

# Copy source
COPY agents/ ./agents/
COPY mcp/ ./mcp/
COPY tools/ ./tools/
COPY Skills.md Agents.md Tools.md ./
COPY task-graph.a2a.json rasic-matrix.json mcp-registry.json ./

# Environment (overridden at runtime)
ENV ANTHROPIC_API_KEY=""
ENV GEMINI_API_KEY=""
ENV FIREBASE_PROJECT="axqxos-platform"
ENV TAPD_EPOCH="Q5"
ENV A2A_MCP_ROOT="http://mcp-relay:8080"
ENV SOVEREIGNTY_CHAIN="RFC8785"
ENV AGENT_MODE="false"
ENV LOG_LEVEL="INFO"

# Expose MCP SSE port
EXPOSE 8080

# Healthcheck
HEALTHCHECK --interval=30s --timeout=10s --start-period=15s --retries=3 \
    CMD curl -f http://localhost:8080/mcp/health || exit 1

CMD ["python", "agents/runner.py"]

# AxQxOS Frontend — Dunder Mifflin OS
# Vite build → Nginx serve
FROM node:20-alpine AS build

WORKDIR /app
COPY frontend/package*.json ./
RUN npm ci

COPY frontend/ .
RUN npm run build

FROM nginx:alpine AS serve
COPY --from=build /app/dist /usr/share/nginx/html
COPY frontend/nginx.conf /etc/nginx/conf.d/default.conf

EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]

FILE: docker-compose.yml

version: "3.9"

services:

  # ── Agent Q Kernel Runner ─────────────────────────────────────────
  kernel:
    build:
      context: .
      dockerfile: Dockerfile.kernel
    container_name: axqxos-kernel
    env_file: .env
    environment:
      - A2A_MCP_ROOT=http://mcp-relay:8080
      - CHROMA_URL=http://vector-store:8000
    depends_on:
      - mcp-relay
      - vector-store
    ports:
      - "8090:8090"
    volumes:
      - ./artifacts:/axqxos/artifacts
      - ./receipts:/axqxos/receipts
    networks:
      - axqxos-mesh
    restart: unless-stopped

  # ── MCP Relay (Echo broker) ────────────────────────────────────────
  mcp-relay:
    build:
      context: .
      dockerfile: Dockerfile.kernel
    container_name: axqxos-mcp-relay
    command: ["python", "mcp/server.py", "--port", "8080"]
    env_file: .env
    ports:
      - "8080:8080"
    networks:
      - axqxos-mesh
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/mcp/health"]
      interval: 20s
      timeout: 5s
      retries: 3

  # ── Gloh Vector Store (Chroma) ─────────────────────────────────────
  vector-store:
    build:
      context: .
      dockerfile: Dockerfile.vector
    container_name: axqxos-vector-store
    env_file: .env
    volumes:
      - chroma-data:/chroma/data
    ports:
      - "8000:8000"
    networks:
      - axqxos-mesh
    restart: unless-stopped

  # ── Frontend (Dunder Mifflin OS) ───────────────────────────────────
  frontend:
    build:
      context: .
      dockerfile: Dockerfile.frontend
    container_name: axqxos-frontend
    ports:
      - "3000:80"
    networks:
      - axqxos-mesh
    restart: unless-stopped
    depends_on:
      - kernel

  # ── Dot CI Runner (GitHub Actions self-hosted, optional) ───────────
  dot-runner:
    image: ghcr.io/actions/actions-runner:latest
    container_name: axqxos-dot-runner
    env_file: .env
    environment:
      - RUNNER_LABELS=axqxos,ubuntu-latest
    volumes:
      - /var/run/docker.sock:/var/run/docker.sock
    networks:
      - axqxos-mesh
    restart: unless-stopped
    profiles: ["ci"]   # opt-in: docker compose --profile ci up

volumes:
  chroma-data:

networks:
  axqxos-mesh:
    driver: bridge
